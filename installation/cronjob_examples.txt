CONFIGURATE POSTFIX (NEEDED URGENTLY TO RUN THIS MODULE) CRONJOB.PHP
-----------------------------------------------------------------
unlink PATHTOBINDFILEINCRONJOBFOLDER; 
cp /etc/bind/named.conf.local PATHTOCRONJOBFOLDERBINDFILE;
chown www-data PATHTOCRONJOBFOLDERBINDFILE;
chgrp www-data PATHTOCRONJOBFOLDERBINDFILE; 
php PATHTOCRONJOBFOLDER/cronjob-autobind.php;


AUTODNS CRONJOB EXAMPLE 
---------------------------------------------
php PATHTOCRONJOB.PHP; 
unlink /etc/postfix/transportmaps; 
unlink /etc/postfix/relaydomains;
mv PATHTOCRONJOBFOLDER/transportmaps /etc/postfix/transportmaps; 
mv PATHTOCRONJOBFOLDER/relaydomains /etc/postfix/relaydomains;
/usr/sbin/postmap /etc/postfix/relaydomains;
/usr/sbin/postmap /etc/postfix/transportmaps;
/etc/init.d/postfix reload
